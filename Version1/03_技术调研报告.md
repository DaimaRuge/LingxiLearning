### 6.1 文档关系图谱构建

```python
# Connected Papers 实现核心逻辑
class CitationGraphBuilder:
    def __init__(self, db_connection):
        self.db = db_connection
        self.graph = None
    
    def build_graph(self, seed_papers):
        """从种子文献构建引用图"""
        # 1. 收集所有参考文献
        all_papers = set(seed_papers)
        queue = list(seed_papers)
        
        while queue:
            paper = queue.pop(0)
            refs = self.fetch_references(paper)
            for ref in refs:
                if ref not in all_papers:
                    all_papers.add(ref)
                    queue.append(ref)
        
        # 2. 构建图
        self.graph = self._create_graph(all_papers)
        
        # 3. 计算节点重要性（基于引用）
        self._calculate_importance()
        
        return self.graph
    
    def find_related(self, seed_id, max_nodes=50):
        """找到最相关的文献"""
        # 使用 BFS + 重要性加权
        scores = {}
        visited = set()
        queue = [(seed_id, 0)]
        
        while queue and len(scores) < max_nodes:
            node, depth = queue.pop(0)
            if node in visited:
                continue
            visited.add(node)
            
            # 计算得分
            score = self.graph.nodes[node].get('importance', 0) / (depth + 1)
            scores[node] = score
            
            # 邻居加入队列
            for neighbor in self.graph.neighbors(node):
                if neighbor not in visited:
                    queue.append((neighbor, depth + 1))
        
        # 返回排序结果
        return sorted(scores.items(), key=lambda x: -x[1])
```

# 多层记忆系统
class PersonalMemory:
    def __init__(self, vector_db, kg_db):
        self.vector_db = vector_db
        self.kg_db = kg_db
        
        # 三层记忆
        self.working_memory = []  # 当前会话
        self.episodic_memory = []  # 短期记忆
        self.semantic_memory = None  # 长期记忆（向量+图谱）
    
    def add(self, content, memory_type="episodic"):
        """添加记忆"""
        if memory_type == "working":
            self.working_memory.append(content)
        elif memory_type == "episodic":
            # 存储到向量数据库
            self.vector_db.insert({
                "text": content,
                "type": "episodic",
                "timestamp": datetime.now()
            })
            self.episodic_memory.append(content)
    
    def retrieve(self, query, top_k=5):
        """检索相关记忆"""
        # 1. 检索向量记忆
        vector_results = self.vector_db.search(query, top_k=top_k)
        
        # 2. 检索图谱关联
        kg_results = self.kg_db.query(query)
        
        # 3. 融合结果
        return self._fuse_results(vector_results, kg_results)
    
    def evolve(self):
        """记忆演化 - 定期整合"""
        # 将高频访问的 episodic 转为 semantic
        frequent = self.vector_db.get_frequent()
        for item in frequent:
            self.semantic_memory.merge(item)
